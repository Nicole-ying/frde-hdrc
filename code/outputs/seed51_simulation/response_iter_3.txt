def compute_reward(obs, action, next_obs, original_reward, info, training_progress=0.0):
    # Extract observation arrays
    o = obs
    n = next_obs
    
    # Observation structure (from environment code):
    # [0]: x position (normalized)
    # [1]: y position (normalized, positive = above helipad)
    # [2]: x velocity (normalized)
    # [3]: y velocity (normalized, positive = upward)
    # [4]: angle (radians)
    # [5]: angular velocity (normalized)
    # [6]: left leg ground contact (0 or 1)
    # [7]: right leg ground contact (0 or 1)
    
    # --- Component 1: Height progress (reward descending toward ground) ---
    # y position decreases as lander descends toward ground
    # Reward negative change in y (descending)
    height_progress = -(n[1] - o[1])  # positive when moving down
    
    # --- Component 2: Vertical velocity penalty (soft landing) ---
    # y velocity (index 3): positive = upward, negative = downward
    # Penalize high downward velocity (negative y velocity)
    vertical_vel_penalty = -min(n[3], 0.0)  # positive when moving down, penalize large magnitude
    
    # --- Component 3: Horizontal velocity penalty ---
    # x velocity (index 2): penalize sideways movement
    horizontal_vel_penalty = -abs(n[2])
    
    # --- Component 4: Stability (angle penalty) ---
    # angle (index 4): penalize being tilted
    angle_penalty = -abs(n[4])
    
    # --- Component 5: Angular velocity penalty ---
    # angular velocity (index 5): penalize spinning
    angular_vel_penalty = -abs(n[5])
    
    # --- Component 6: Action cost ---
    action_cost = 0.0
    if action == 2:  # Main engine
        action_cost = 0.5
    elif action in [1, 3]:  # Side engines
        action_cost = 0.2
    
    # --- Component 7: Ground contact reward ---
    ground_contact = o[6] + o[7]  # sum of leg contacts
    
    # --- Component 8: Survival bonus ---
    survival_bonus = 0.1  # small constant reward for staying alive
    
    # Stage-based weights
    if training_progress < 0.3:
        # Early training: focus on exploration, descending, and basic stability
        w_height = 1.0
        w_vert_vel = -0.2
        w_horiz_vel = -0.1
        w_angle = -0.3
        w_ang_vel = -0.1
        w_action = -0.1
        w_contact = 0.5
        w_survival = 0.1
    elif training_progress < 0.7:
        # Middle training: balance all components
        w_height = 0.8
        w_vert_vel = -0.5
        w_horiz_vel = -0.3
        w_angle = -0.5
        w_ang_vel = -0.2
        w_action = -0.3
        w_contact = 1.0
        w_survival = 0.2
    else:
        # Late training: emphasize precision landing
        w_height = 0.5
        w_vert_vel = -1.0
        w_horiz_vel = -0.5
        w_angle = -1.0
        w_ang_vel = -0.5
        w_action = -0.5
        w_contact = 2.0
        w_survival = 0.3
    
    # Combine components
    reward = (w_height * height_progress +
              w_vert_vel * vertical_vel_penalty +
              w_horiz_vel * horizontal_vel_penalty +
              w_angle * angle_penalty +
              w_ang_vel * angular_vel_penalty +
              w_action * action_cost +
              w_contact * ground_contact +
              w_survival * survival_bonus)
    
    return reward
