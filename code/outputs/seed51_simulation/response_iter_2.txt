def compute_reward(obs, action, next_obs, original_reward, info, training_progress=0.0):
    # Extract observation arrays
    o = obs
    n = next_obs
    
    # --- Component 1: Progress toward goal (directional) ---
    # Use position y (index 1) to reward descent toward ground
    # Positive when moving downward (closer to ground)
    height_progress = o[1] - n[1]  # positive when descending
    
    # --- Component 2: Stability - penalize angle and angular velocity ---
    # Angle is index 4, angular velocity is index 5
    angle_penalty = -abs(n[4])  # penalize being tilted
    angular_vel_penalty = -abs(n[5])  # penalize spinning
    
    # --- Component 3: Velocity penalty for soft landing ---
    # Vertical velocity is index 3, horizontal velocity is index 2
    vertical_vel_penalty = -abs(n[3])  # penalize vertical speed
    horizontal_vel_penalty = -abs(n[2])  # penalize horizontal speed
    
    # --- Component 4: Action cost ---
    action_cost = 0.0
    if action == 2:  # Main engine
        action_cost = 0.5
    elif action in [1, 3]:  # Side engines
        action_cost = 0.2
    
    # --- Component 5: Ground contact bonus ---
    ground_contact = o[-1] + o[-2] if len(o) >= 2 else 0.0
    
    # --- Component 6: Survival bonus ---
    survival_bonus = 1.0  # small constant reward for staying alive
    
    # Stage-based weights
    if training_progress < 0.3:
        # Early: focus on exploration, height progress, and stability
        w_height = 2.0
        w_angle = -0.5
        w_angular_vel = -0.3
        w_vert_vel = -0.2
        w_horiz_vel = -0.1
        w_action = -0.3
        w_contact = 0.5
        w_survival = 0.5
    elif training_progress < 0.7:
        # Middle: balance all components
        w_height = 1.5
        w_angle = -1.0
        w_angular_vel = -0.5
        w_vert_vel = -0.5
        w_horiz_vel = -0.3
        w_action = -0.5
        w_contact = 1.0
        w_survival = 0.8
    else:
        # Late: emphasize precision landing and soft touchdown
        w_height = 1.0
        w_angle = -2.0
        w_angular_vel = -1.0
        w_vert_vel = -1.0
        w_horiz_vel = -0.5
        w_action = -0.7
        w_contact = 2.0
        w_survival = 1.0
    
    # Combine components
    reward = (w_height * height_progress +
              w_angle * angle_penalty +
              w_angular_vel * angular_vel_penalty +
              w_vert_vel * vertical_vel_penalty +
              w_horiz_vel * horizontal_vel_penalty +
              w_action * action_cost +
              w_contact * ground_contact +
              w_survival * survival_bonus)
    
    return reward
