def compute_reward(obs, action, next_obs, original_reward, info, training_progress=0.0):
    # Extract observation arrays
    o = obs
    n = next_obs
    
    # --- Component 1: Progress toward center (directional) ---
    # Use position elements (indices 0,1) to encourage moving toward center
    progress = (abs(o[0]) - abs(n[0])) + (abs(o[1]) - abs(n[1]))
    
    # --- Component 2: Velocity penalty (encourage soft landing) ---
    # Velocity elements are at indices 2,3
    vel_penalty = abs(n[2]) + abs(n[3])
    
    # --- Component 3: Stability (angle and angular velocity) ---
    # Angle at index 4, angular velocity at index 5
    angle_penalty = abs(n[4]) + abs(n[5])
    
    # --- Component 4: Action cost ---
    action_cost = 0.0
    if action == 2:  # Main engine
        action_cost = 0.5
    elif action in [1, 3]:  # Side engines
        action_cost = 0.2
    
    # --- Component 5: Ground contact bonus ---
    ground_contact = n[-1] + n[-2] if len(n) >= 2 else 0.0
    
    # --- Component 6: Survival bonus (alive each step) ---
    survival = 1.0
    
    # Stage-based weights
    if training_progress < 0.3:
        # Early: explore, learn to move, stay alive
        w_progress = 2.0
        w_vel = -0.5
        w_angle = -0.3
        w_action = -0.1
        w_contact = 0.5
        w_survival = 1.0
    elif training_progress < 0.7:
        # Middle: balance movement with control
        w_progress = 1.5
        w_vel = -1.0
        w_angle = -0.8
        w_action = -0.3
        w_contact = 1.5
        w_survival = 1.0
    else:
        # Late: precision landing
        w_progress = 1.0
        w_vel = -2.0
        w_angle = -1.5
        w_action = -0.5
        w_contact = 3.0
        w_survival = 1.0
    
    # Combine components
    reward = (w_progress * progress +
              w_vel * vel_penalty +
              w_angle * angle_penalty +
              w_action * action_cost +
              w_contact * ground_contact +
              w_survival * survival)
    
    return reward
