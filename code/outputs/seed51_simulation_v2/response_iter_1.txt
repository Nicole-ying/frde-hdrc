def compute_reward(obs, action, next_obs, original_reward, info, training_progress=0.0):
    # Extract observation arrays
    o = obs
    n = next_obs
    
    # Feature 1: Change in absolute values (encourages movement toward desirable states)
    abs_diff = sum(abs(o[i]) - abs(n[i]) for i in range(len(o)))
    
    # Feature 2: Stability - penalize angle and angular velocity
    # obs[4] is angle, obs[5] is angular velocity
    angle_penalty = abs(o[4]) - abs(n[4])  # directional: reducing angle is good
    ang_vel_penalty = abs(n[5])  # penalize high angular velocity
    
    # Feature 3: Velocity penalty - penalize high horizontal and vertical velocity
    # obs[2] is horizontal velocity, obs[3] is vertical velocity
    hvel_penalty = abs(n[2])  # penalize high horizontal velocity
    vvel_penalty = abs(n[3])  # penalize high vertical velocity
    
    # Feature 4: Action penalty (discourages excessive action usage)
    action_cost = 0.0
    if action == 2:  # Main engine
        action_cost = 0.5
    elif action in [1, 3]:  # Side engines
        action_cost = 0.2
    
    # Feature 5: Ground contact bonus (from info or observation)
    ground_contact = o[-1] + o[-2] if len(o) >= 2 else 0.0
    
    # Stage-based weights
    if training_progress < 0.3:
        # Early training: encourage exploration and basic stability
        w_abs = 1.0
        w_angle = 0.5
        w_ang_vel = -0.3
        w_hvel = -0.2
        w_vvel = -0.3
        w_action = -0.2
        w_contact = 0.5
    elif training_progress < 0.7:
        # Middle training: balance all objectives
        w_abs = 0.8
        w_angle = 0.8
        w_ang_vel = -0.5
        w_hvel = -0.4
        w_vvel = -0.5
        w_action = -0.4
        w_contact = 1.0
    else:
        # Late training: precision landing
        w_abs = 0.5
        w_angle = 1.0
        w_ang_vel = -0.7
        w_hvel = -0.6
        w_vvel = -0.7
        w_action = -0.6
        w_contact = 2.0
    
    # Combine components
    reward = (w_abs * abs_diff + 
              w_angle * angle_penalty + 
              w_ang_vel * ang_vel_penalty + 
              w_hvel * hvel_penalty + 
              w_vvel * vvel_penalty + 
              w_action * action_cost + 
              w_contact * ground_contact)
    
    return reward
