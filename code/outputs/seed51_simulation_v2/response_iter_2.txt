def compute_reward(obs, action, next_obs, original_reward, info, training_progress=0.0):
    # Extract observation arrays
    o = obs
    n = next_obs
    
    # Observation structure (BlackBoxControlEnv):
    # [pos_x, pos_y, vel_x, vel_y, angle, angular_vel, leg1_contact, leg2_contact]
    
    # Feature 1: Change in absolute values (encourages movement toward desirable states)
    # Focus on position (indices 0,1) and velocity (indices 2,3)
    abs_diff = sum(abs(o[i]) - abs(n[i]) for i in range(4))
    
    # Feature 2: Stability - penalize large angle and angular velocity
    # Angle is index 4, angular velocity is index 5
    angle_penalty = abs(o[4]) + abs(o[5])  # Penalize current angle and angular velocity
    
    # Feature 3: Velocity penalty - penalize high horizontal/vertical velocity
    # vel_x is index 2, vel_y is index 3
    velocity_penalty = abs(o[2]) + abs(o[3])
    
    # Feature 4: Action penalty (discourages excessive action usage)
    action_cost = 0.0
    if action == 2:  # Main engine
        action_cost = 0.5
    elif action in [1, 3]:  # Side engines
        action_cost = 0.2
    
    # Feature 5: Ground contact bonus (from observation)
    # Last two elements of obs are ground contact indicators
    ground_contact = o[-1] + o[-2] if len(o) >= 2 else 0.0
    
    # Stage-based weights
    # Early stage: focus on exploration and smooth movement
    # Middle stage: balance movement and control
    # Late stage: emphasize precision and ground contact
    
    if training_progress < 0.3:
        # Early training: encourage movement and exploration
        w_abs = 1.0
        w_angle = -0.5
        w_vel = -0.3
        w_action = -0.2
        w_contact = 0.5
    elif training_progress < 0.7:
        # Middle training: balance
        w_abs = 0.8
        w_angle = -0.8
        w_vel = -0.5
        w_action = -0.4
        w_contact = 1.0
    else:
        # Late training: precision and landing
        w_abs = 0.5
        w_angle = -1.0
        w_vel = -0.8
        w_action = -0.6
        w_contact = 2.0
    
    # Combine components
    reward = (w_abs * abs_diff + 
              w_angle * angle_penalty + 
              w_vel * velocity_penalty + 
              w_action * action_cost + 
              w_contact * ground_contact)
    
    return reward
