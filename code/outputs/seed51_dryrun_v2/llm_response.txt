def compute_reward(obs, action, next_obs, original_reward, info, training_progress=0.0):
    # Extract observation arrays
    o = obs
    n = next_obs
    
    # Feature 1: Directional progress - encourage moving toward desirable states
    # Use difference in absolute values (directional: positive when moving toward zero)
    directional_progress = sum(abs(n[i]) - abs(o[i]) for i in range(len(o)))
    
    # Feature 2: Velocity penalty - discourage high speeds for stability
    # Observations 2 and 3 are velocity components (x and y)
    velocity_penalty = 0.0
    if len(o) >= 4:
        current_vel = (o[2]**2 + o[3]**2)**0.5
        next_vel = (n[2]**2 + n[3]**2)**0.5
        velocity_penalty = next_vel  # Penalize high velocity
    
    # Feature 3: Stability - encourage upright orientation
    # Observation 4 is angle, observation 5 is angular velocity
    angle_penalty = 0.0
    angular_vel_penalty = 0.0
    if len(o) >= 6:
        angle_penalty = abs(n[4])  # Penalize deviation from upright
        angular_vel_penalty = abs(n[5])  # Penalize angular velocity
    
    # Feature 4: Action cost - discourage excessive engine usage
    action_cost = 0.0
    if action == 2:  # Main engine
        action_cost = 1.0
    elif action in [1, 3]:  # Side engines
        action_cost = 0.5
    
    # Feature 5: Ground contact bonus (from next_obs for proper timing)
    ground_contact = 0.0
    if len(n) >= 2:
        ground_contact = n[-1] + n[-2] if len(n) >= 2 else 0.0
    
    # Feature 6: Survival bonus - reward for staying alive
    survival_bonus = 1.0
    
    # Stage-based weights
    if training_progress < 0.3:
        # Early: explore, learn to move, avoid crashing
        w_directional = 0.3
        w_velocity = -0.4
        w_angle = -0.5
        w_angular_vel = -0.3
        w_action = -0.2
        w_contact = 0.5
        w_survival = 1.0
    elif training_progress < 0.7:
        # Middle: balance movement with stability
        w_directional = 0.5
        w_velocity = -0.6
        w_angle = -0.8
        w_angular_vel = -0.5
        w_action = -0.4
        w_contact = 1.0
        w_survival = 1.5
    else:
        # Late: precision landing
        w_directional = 0.7
        w_velocity = -0.8
        w_angle = -1.0
        w_angular_vel = -0.6
        w_action = -0.6
        w_contact = 2.0
        w_survival = 2.0
    
    # Combine components
    reward = (w_directional * directional_progress +
              w_velocity * velocity_penalty +
              w_angle * angle_penalty +
              w_angular_vel * angular_vel_penalty +
              w_action * action_cost +
              w_contact * ground_contact +
              w_survival * survival_bonus)
    
    return reward
